int f1(int a[], int len)
{

    {

int f2(int n)
{
    if (n == 1)
        return 1;
    else if (n == 2)
        return 1;
    else
        return f2(n - 1) + f2(n - 2);
}

int f4(int a[], int b[], int len)
{
    int i, sum = 0;
    for (i = 0; i < len; i = i + 1)
